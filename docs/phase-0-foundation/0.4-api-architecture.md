6. Custom Cloud Functions
These will be deployed to Firebase Cloud Functions to handle server-side logic.
Update Location Rating
Trigger: When a new review is created
Function: Recalculate location's average rating
javascript// Cloud Function (Node.js)
exports.updateLocationRating = functions.firestore
  .document('reviews/{reviewId}')
  .onCreate(async (snap, context) => {
    const review = snap.data();
    const locationRef = db.collection('locations').doc(review.locationId);
    
    const reviews = await db.collection('reviews')
      .where('locationId', '==', review.locationId)
      .get();
    
    const totalRating = reviews.docs.reduce((sum, doc) => 
      sum + doc.data().rating, 0
    );
    const avgRating = totalRating / reviews.size;
    
    await locationRef.update({
      averageRating: avgRating,
      totalReviews: reviews.size
    });
  });

Standard Error Response Format
javascript{
  success: false,

    message: "string (Human-readable message)",
    details: {} // Optional additional info
  }
}
Common Error Codes
javascriptconst ERROR_CODES = {
  // Authentication
  AUTH_INVALID_CREDENTIALS: "Invalid email or password",
  AUTH_USER_NOT_FOUND: "User not found",
  AUTH_EMAIL_IN_USE: "Email already registered",
  
  // Authorization
  PERMISSION_DENIED: "You don't have permission to perform this action",
  UNAUTHORIZED: "Please log in to continue",
  
  // Data
  NOT_FOUND: "Resource not found",
  INVALID_INPUT: "Invalid input data",
  DUPLICATE_ENTRY: "This entry already exists",
  
  // External APIs
  MAPS_API_ERROR: "Error fetching map data",
  OPENAI_API_ERROR: "AI service temporarily unavailable",
  
  // Network
  NETWORK_ERROR: "Network connection failed",
  TIMEOUT: "Request timed out",
  
  // Rate Limiting
  RATE_LIMIT_EXCEEDED: "Too many requests, please try again later"
};
Error Handling Wrapper
javascript// Utility function for API calls
async function handleApiCall(apiFunction) {
  try {
    const result = await apiFunction();
    return { success: true, data: result };
  } catch (error) {
    console.error('API Error:', error);
    
    // Map Firebase errors
    if (error.code?.startsWith('auth/')) {
      return {
        success: false,
        error: {
          code: error.code,
          message: ERROR_CODES[error.code] || error.message
        }
      };
    }
    
    // Generic error
    return {
      success: false,
      error: {
        code: 'UNKNOWN_ERROR',
        message: 'An unexpected error occurred'
      }
    };
  }
}
